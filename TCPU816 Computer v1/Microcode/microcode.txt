Note: All '$nn' or '$nnnn' arguments may be replaced by a label name.

NOP -  No operation
	0x00: NONE    	-	 NOP 

EDO -  Write ACC to external device.
	0x01: IND    	-	 EDO $nnnn
	0x02: DIND    	-	 EDO ($nnnn)

	0x03: IND    	-	 EDI $nnnn
	0x04: DIND    	-	 EDI ($nnnn)

	0x07: DIND    	-	 LDA ($nnnn)
	0x08: DINDX    	-	 LDA ($nnnn),X
	0x09: DINDY    	-	 LDA ($nnnn),Y
	0x0A: INDX    	-	 LDA $nnnn,X
	0x0B: INDY    	-	 LDA $nnnn,Y

	0x0C: IMM    	-	 LDX #$nn
	0x0D: IND    	-	 LDX $nnnn

	0x0E: IMM    	-	 LDY #$nn
	0x0F: IND    	-	 LDY $nnnn

	0x15: INDY    	-	 STA $nnnn,Y
